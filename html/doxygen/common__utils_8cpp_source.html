18 bool is_dir_exist(const std::string& path) {
19  struct stat info {};
20  if (stat(path.c_str(), &info) != 0) {
21  return false;
22  }
23  return (info.st_mode & S_IFDIR) != 0;

26 bool make_path(const std::string& path) {
27  mode_t mode = 0755;
28  int ret = mkdir(path.c_str(), mode);
29  if (ret == 0) {
30  return true;
31  }
32 
33  switch (errno) {
34  case ENOENT:
35  // parent didn't exist, try to create it
36  {
37  int pos = path.find_last_of('/');
38  if (pos == std::string::npos) {
39  return false;
40  }
41  if (!make_path(path.substr(0, pos))) {
42  return false;
43  }
44  }
45  // now, try to create again
46  return 0 == mkdir(path.c_str(), mode);
47 
48  case EEXIST:
49  // done!
50  return is_dir_exist(path);
51 
52  default:
53  auto msg = "Can not create directory " + path;
54  throw std::runtime_error(msg);
55  }
56 }
